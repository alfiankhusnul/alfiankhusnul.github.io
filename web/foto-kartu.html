const video = document.getElementById('video'); const canvas = document.getElementById('canvas'); const captureBtn = document.getElementById('captureBtn'); const downloadBtn = document.getElementById('downloadBtn'); const downloadBtnTag = document.getElementById('downloadBtnTag'); const previewImg = document.getElementById('previewImg'); const previewArea = document.getElementById('previewArea'); const fileSizeInfo = document.getElementById('fileSizeInfo'); // Gunakan resolusi internal lebih tinggi untuk ketajaman (4x dari 300x200) const TARGET_WIDTH = 1200; const TARGET_HEIGHT = 800; async function initCamera() { try { const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment", width: { ideal: 1920 }, // Minta resolusi Full HD dari sensor height: { ideal: 1080 } }, audio: false }); video.srcObject = stream; } catch (err) { alert("Akses kamera ditolak atau tidak tersedia."); } } captureBtn.addEventListener('click', () => { // Set ukuran canvas ke resolusi tinggi sebelum render canvas.width = TARGET_WIDTH; canvas.height = TARGET_HEIGHT; const ctx = canvas.getContext('2d'); // Aktifkan image smoothing untuk kualitas maksimal ctx.imageSmoothingEnabled = true; ctx.imageSmoothingQuality = 'high'; const videoWidth = video.videoWidth; const videoHeight = video.videoHeight; // Kalkulasi rasio untuk cropping tengah const videoRatio = videoWidth / videoHeight; const targetRatio = TARGET_WIDTH / TARGET_HEIGHT; let drawWidth, drawHeight, startX, startY; if (videoRatio > targetRatio) { drawHeight = videoHeight; drawWidth = videoHeight * targetRatio; startX = (videoWidth - drawWidth) / 2; startY = 0; } else { drawWidth = videoWidth; drawHeight = videoWidth / targetRatio; startX = 0; startY = (videoHeight - drawHeight) / 2; } // Gambar ke canvas resolusi tinggi ctx.drawImage(video, startX, startY, drawWidth, drawHeight, 0, 0, TARGET_WIDTH, TARGET_HEIGHT); // Kompresi dengan kualitas maksimal (1.0) // Karena target 1200x800, sizenya akan berkisar 100kb - 180kb (Masuk target < 200kb) let quality = 0.95; let dataUrl = canvas.toDataURL('image/jpeg', quality); // Tampilkan di UI (tetap terlihat kecil 300x200 di CSS agar rapi) previewImg.src = dataUrl; previewArea.style.display = 'block'; downloadBtn.style.display = 'block'; const sizeInBytes = Math.round((dataUrl.split(',')[1].length * 3 / 4)); const sizeInKB = (sizeInBytes / 1024).toFixed(2); fileSizeInfo.innerText = `Ukuran File: ${sizeInKB} KB | Resolusi: ${TARGET_WIDTH}x${TARGET_HEIGHT}`; downloadBtnTag.href = dataUrl; downloadBtnTag.download = `ID_Sharp_${Date.now()}.jpg`; }); initCamera();
